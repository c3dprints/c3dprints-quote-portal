<!-- 3) Add these JS functions near your other JS functions: -->

<script>
function statusClass(status){
  return "status-"+String(status||"New").replaceAll(" ","-");
}

async function updateStatus(){
  if(!selectedRequest) return;

  $("status-message").textContent="";
  $("status-error").textContent="";

  const newStatus=$("status-select").value;

  try{
    const response=await apiFetch(`${API_BASE}/admin/requests/${selectedRequest.id}/status`,{
      method:"PATCH",
      body:JSON.stringify({status:newStatus})
    });

    if(!response.ok){
      const problem=await response.json().catch(()=>({detail:"Status update failed"}));
      throw new Error(problem.detail||"Status update failed");
    }

    const data=await response.json();
    const idx=requests.findIndex(r=>r.id===selectedRequest.id);

    if(idx>=0) requests[idx]=data.request;

    selectedRequest=data.request;
    $("d-status").textContent=data.request.status;
    $("status-select").value=data.request.status;
    $("status-message").textContent="Status saved.";

    renderRequests();
  }catch(error){
    $("status-error").textContent=error.message||"Status update failed";
  }
}
</script>
